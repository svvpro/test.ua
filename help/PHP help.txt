//Включить отображение ошибок
ini_set('display_errors', 1);
error_reporting(E_ALL);

$_SERVER['REQUEST_URI'] - получаем строку запроса

//Настройки htaccess
AddDefaultCharset utf-8 - устанавливает кодировку по умолчанию
RewriteEngine on - включает функфионал модуля mode_rewrite
RewriteBase / -
RewriteRule ^(.*)$ index.php - перенаправляет все запросы на файл index.php

//Функции
var_dump() — Выводит информацию о переменной
print_r() — Выводит удобочитаемую информацию о переменной
dirname() - возвращает имя каталога из указанного пути
preg_match() - выполняет проверку на соответствие регулярному выражению
explode() — разбивает строку с помощью разделителя
implode() — объединяет элементы массива в строку
ucfirst() — преобразует первый символ строки в верхний регистр
file_exists() — Проверяет наличие указанного файла или каталога
empty() — Проверяет, пуста ли переменная
array_shift() извлекает первое значение массива array и возвращает его, сокращая размер array на один элемент.

//Константы
__FILE__ - полный путь и имя текущего файла
__DIR__ - директория файла. Если используется внутри подключаемого файла, то возвращается директория этого файла. Это эквивалентно вызову dirname(__FILE__)

//План
    FrontController
    1)Общие настройки
    2)Подключение файловой системы
    3)Установка соеденения с БД
    ROUTER
    4)Анализ запроса определение контроллера
    5)Подключение контроллера
    6)Передача управления контроллеру
